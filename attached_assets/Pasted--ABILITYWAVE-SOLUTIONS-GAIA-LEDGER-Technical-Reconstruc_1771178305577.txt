model CropSheet {
  cropSheetId      Int               @id @default(autoincrement())

model HealthStatusType {
  statusId      Int               @id @default(autoincrement())

  displayName   String            // User-facing name (e.g., "Healthy")
  color         String            @default("#22c55e")
  cropSheet     CropSheet?        @relation(fields: [cropSheetId], references: [cropSheetId])
  cropSheetId   Int?
  healthCurrent HealthCurrent[]
  healthHistory HealthHistory[]

  @@map("health_status_types")

model Row {
  rowId    Int     @id @default(autoincrement())
  letter   String  @unique
  plants   Plant[]

  @@map("rows")

model Plant {
  plantId       Int                      @id @default(autoincrement())
  code          String                   @unique // e.g., "A1"
  number        Int
  plantedDate   DateTime?
  status        String                   @default("Active")
  
  rowId         Int
  row           Row                      @relation(fields: [rowId], references: [rowId])
  
  cropSheetId   Int?
  cropSheet     CropSheet?               @relation(fields: [cropSheetId], references: [cropSheetId])

  currentHealth HealthCurrent?
  history       HealthHistory[]
  currentAttrs  PlantAttributeCurrent[]
  attrHistory   PlantAttributeHistory[]

  @@map("plants")
}

model HealthCurrent {
  id           Int      @id @default(autoincrement())
  plantId      Int      @unique
  plant        Plant    @relation(fields: [plantId], references: [plantId], onDelete: Cascade)
  statusId     Int
  status       HealthStatusType @relation(fields: [statusId], references: [statusId])
  notes        String?
  lastChecked  DateTime @default(now())

  @@map("plant_health_current")

model HealthHistory {
  id           Int      @id @default(autoincrement())
  plantId      Int
  plant        Plant    @relation(fields: [plantId], references: [plantId], onDelete: Cascade)
  statusId     Int
  status       HealthStatusType @relation(fields: [statusId], references: [statusId])
  notes        String?
  checkedAt    DateTime @default(now())

  @@map("plant_health_history")

model Attribute {
  attributeId  Int      @id @default(autoincrement())

4.2 Crop Sheets API (The Remediation Logic)
File: src/app/api/crop-sheets/route.ts
This file handles the "Unique Name" fix to prevent database collisions.

import { NextRequest, NextResponse } from 'next/server'
import { db } from '@/lib/db'

// GET - Fetch all crop sheets
export async function GET() {
  try {
    const cropSheets = await db.cropSheet.findMany({
      include: {
        healthStatuses: {
          orderBy: { statusId: 'asc' }
        }
      },
      orderBy: { createdAt: 'desc' }
    })
    return NextResponse.json(cropSheets)
  } catch (error) {
    console.error('Error fetching crop sheets:', error)
    return NextResponse.json({ error: 'Failed to fetch crop sheets', code: 'CROPSHEET_GET_500' }, { status: 500 })
  }
}

// POST - Create new crop sheet with default health statuses
export async function POST(request: NextRequest) {
  try {
    const { name, description } = await request.json()

    if (!name || typeof name !== 'string' || !name.trim()) {
      return NextResponse.json({ error: 'Name is required', code: 'CROPSHEET_001' }, { status: 400 })
    }

    // Check duplicate name
    const existing = await db.cropSheet.findUnique({ where: { name: name.trim() } })
    if (existing) {
      return NextResponse.json({ error: 'Crop sheet already exists', code: 'CROPSHEET_002' }, { status: 409 })
    }

    // Templates with correct colors
    const templates = [
      { name: 'Healthy', color: '#22c55e' },           // Green
      { name: 'Dehydrated', color: '#eab308' },        // Yellow
      { name: 'Fungus', color: '#78350f' },            // Dark Brown
      { name: 'Pest Damage', color: '#ef4444' },       // Red
      { name: 'Nutrient Deficiency', color: '#f97316' }, // Orange
      { name: 'Not Checked', color: '#6b7280' }        // Gray
    ]

    // Create Crop Sheet
    const cropSheet = await db.cropSheet.create({
      data: {
        name: name.trim(),
        description: description?.trim() || null,
        isActive: true,
      }
    })

    // Create unique health statuses
    const timestamp = Date.now()
    for (const template of templates) {
      // Unique internal name to avoid DB collision
      const uniqueName = `${template.name}_${cropSheet.cropSheetId}_${timestamp}`
      
      await db.healthStatusType.create({
        data: {
          name: uniqueName,
          displayName: template.name,
          color: template.color,
          cropSheet: { connect: { cropSheetId: cropSheet.cropSheetId } }
        }
      })
    }

    return NextResponse.json(cropSheet, { status: 201 })

  } catch (error: any) {
    if (error.code === 'P2002') {
      return NextResponse.json({ error: 'Conflict creating crop sheet', code: 'CROPSHEET_002' }, { status: 409 })
    }
    return NextResponse.json({ error: 'Internal server error', code: 'CROPSHEET_500' }, { status: 500 })
  }
}

6.2 Service Worker (public/sw.js)
Note: Current logs say offline is NOT implemented, but to make it a true PWA, this structure is needed:

const CACHE_NAME = 'gaia-ledger-v1';
const urlsToCache = ['/', '/api/rows', '/api/plants', '/api/health'];

self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response => response || fetch(event.request))
  );
});
